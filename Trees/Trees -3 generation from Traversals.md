### Counting Nodes and height
```
int countNodes(Node *pNode){
    if(pNode == NULL)
        return 0;
    else
        return countNodes(pNode->lchild) + countNodes(pNode->rchild) +1;
}

int heightTree(Node *pNode){
    int x = 0, y = 0;
    if(pNode == NULL)
        return 0;
    if(pNode->lchild){
        x = heightTree(pNode->lchild) +1;
    }
    if(pNode->rchild){
        x = heightTree(pNode->rchild) +1;
    }
    if(x > y)
        return x;
    else
        return y;
}
```
